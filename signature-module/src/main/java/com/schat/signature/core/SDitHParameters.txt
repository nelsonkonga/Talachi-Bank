package com.schat.signature.core;

public class SDitHParameters {
    // Security levels..
    public static final int SECURITY_LEVEL_128 = 128;
    public static final int SECURITY_LEVEL_192 = 192;
    public static final int SECURITY_LEVEL_256 = 256;

    private final int securityLevel;
    private final int n; // Code length..
    private final int k; // Code dimension..
    private final int w; // Syndrome weight..
    private final int tau; // Number of MPC parties..

    public SDitHParameters(int securityLevel) {
        this.securityLevel = securityLevel;

        // Parameters based on security level (simplified example)..
        switch (securityLevel) {
            case SECURITY_LEVEL_128:
                this.n = 230;
                this.k = 126;
                this.w = 79;
                this.tau = 16;
                break;
            case SECURITY_LEVEL_192:
                this.n = 352;
                this.k = 198;
                this.w = 120;
                this.tau = 16;
                break;
            case SECURITY_LEVEL_256:
                this.n = 480;
                this.k = 274;
                this.w = 161;
                this.tau = 16;
                break;
            default:
                throw new IllegalArgumentException("Unsupported security level");
        }
    }

    // Getters..
    public int getSecurityLevel() { return securityLevel; }
    public int getN() { return n; }
    public int getK() { return k; }
    public int getW() { return w; }
    public int getTau() { return tau; }
}
